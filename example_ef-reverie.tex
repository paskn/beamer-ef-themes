\begin{frame}[label={sec:org520625d}]{What Are Prime Numbers?}
\begin{block}{What Are Prime Numbers?}
A \alert{prime number} is a number that has exactly two divisors.
\end{block}
\begin{block}{Example}
\begin{itemize}
\item 2 is prime (two divisors: 1 and 2). \pause
\item 3 is prime (two divisors: 1 and 3). \pause
\item 4 is not prime (\alert{three} divisors: 1, 2, and 4).
\end{itemize}
\end{block}
\end{frame}
\section{Results}
\label{sec:orgf216b16}

\begin{frame}[label={sec:orgac09976}]{There Is No Largest Prime Number}
\framesubtitle{The proof uses \textit{reductio ad absurdum}.}

\begin{theorem}
There is no largest prime number.
\end{theorem}

\begin{proof}
\begin{enumerate}
\item<1-> Suppose \(p\) were the largest prime number.
\item<2-> Let \(q\) be the product of the first \(p\) numbers.
\item<3-> Then \(q + 1\) is not divisible by any of them.
\item<4-> Thus \(q + 1\) is also prime and greater than \(p\).\qedhere
\end{enumerate}
\end{proof}

\uncover<4->{The proof used \textit{reductio ad absurdum}.}
\end{frame}
